DECLARE v_start_date DATE := DATE '2025-07-01'; v_end_date DATE := TRUNC(LAST_DAY(SYSDATE)); v_mo_count PLS_INTEGER; v_mo_date DATE; v_pi PLS_INTEGER; v_pmax PLS_INTEGER := 0; v_gtot NUMBER := 0; V_BAGS_TYPE VARCHAR2(100); V_ITEM_NAME VARCHAR2(100); v_mi VARCHAR2(10000); -- Simple typed arrays TYPE arr_vc IS TABLE OF VARCHAR2(500) INDEX BY PLS_INTEGER; TYPE arr_num IS TABLE OF NUMBER INDEX BY PLS_INTEGER; a_pid arr_vc; -- product ID a_pname arr_vc; -- product name a_ptotal arr_num; -- product grand total -- Flat bag matrix: product_idx * 200 + month_idx a_bags arr_num; a_motot arr_num; -- monthly grand totals BEGIN -- Get product name for selected item BEGIN SELECT LISTAGG( ITM.ITEM_NAME || ' ' || ITM.PACKING_SIZE || '('|| ITM.UNIT ||') '|| ITM.PACKING, ', ' ) WITHIN GROUP (ORDER BY ITM.ITEM_ID) INTO V_ITEM_NAME FROM AB_ITEMS_MASTER ITM WHERE ITM.ITEM_ID IN ( SELECT REGEXP_SUBSTR(:P1171_ITEM_NAME, '[^:]+', 1, LEVEL) FROM dual CONNECT BY REGEXP_SUBSTR(:P1171_ITEM_NAME, '[^:]+', 1, LEVEL) IS NOT NULL ); EXCEPTION WHEN NO_DATA_FOUND THEN V_ITEM_NAME := 'All Products'; END; -- Determine bags type IF :P1171_BAGS_TYPE IS NULL THEN V_BAGS_TYPE := 'All Types'; ELSIF INSTR(:P1171_BAGS_TYPE, 'Normal Sale Order') > 0 AND INSTR(:P1171_BAGS_TYPE, 'Advance Sale Order') > 0 THEN V_BAGS_TYPE := 'Normal Sale Order & Advance Sale Order'; ELSIF :P1171_BAGS_TYPE = 'Normal Sale Order' THEN V_BAGS_TYPE := 'Normal Sale Order'; ELSIF :P1171_BAGS_TYPE = 'Advance Sale Order' THEN V_BAGS_TYPE := 'Advance Sale Order'; ELSE V_BAGS_TYPE := 'All Types'; END IF; v_mo_count := MONTHS_BETWEEN( TRUNC(v_end_date, 'MM'), TRUNC(v_start_date, 'MM') ) + 1; -- Initialize monthly grand-total slots FOR i IN 0 .. v_mo_count - 1 LOOP a_motot(i) := 0; END LOOP; -- Single query: one row per (product, month) FOR r IN ( WITH USER_APPROVAL AS( SELECT APP_IDS APP_ID FROM AB_USER_ACTION_APPROVAL WHERE APP_TYPE = '820' AND STATUS = 'Y' AND APPROVAL_STATUS = 'APPROVED' ), SHIPMENT_DATA AS ( SELECT C.ITEM_ID, C.PRODUCT_NAME, NVL(C.NO_BAGS, 0) NO_BAGS, TRUNC(NVL(C.SHIPMENT_DATE, C.ADD_DATE), 'MM') SHIP_MONTH, STATUS FROM TABLE(AB_SALE_ORDER_PKG.SALE_ORDER_SHIPMENT(:GV_ORG_ID)) C JOIN USER_APPROVAL UA ON UA.APP_ID = C.SHIPMENT_IDD WHERE TRUNC(NVL(C.SHIPMENT_DATE, C.ADD_DATE)) BETWEEN v_start_date AND v_end_date ) SELECT ITEM_ID, PRODUCT_NAME, MONTHS_BETWEEN( TRUNC(SHIP_MONTH, 'MM'), TRUNC(v_start_date, 'MM') ) m_idx, SUM(NO_BAGS) bags FROM SHIPMENT_DATA WHERE (ITEM_ID IN ( SELECT REGEXP_SUBSTR(:P1171_ITEM_NAME, '[^:]+', 1, LEVEL) FROM dual CONNECT BY REGEXP_SUBSTR(:P1171_ITEM_NAME, '[^:]+', 1, LEVEL) IS NOT NULL ) OR :P1171_ITEM_NAME IS NULL) AND ( :P1171_BAGS_TYPE IS NULL OR UPPER(TRIM(STATUS)) IN ( SELECT UPPER(TRIM(REGEXP_SUBSTR(:P1171_BAGS_TYPE, '[^:]+', 1, LEVEL))) FROM dual CONNECT BY REGEXP_SUBSTR(:P1171_BAGS_TYPE, '[^:]+', 1, LEVEL) IS NOT NULL ) ) GROUP BY ITEM_ID, PRODUCT_NAME, SHIP_MONTH ORDER BY PRODUCT_NAME ) LOOP v_mi := r.m_idx; -- Find existing product slot, or create new one v_pi := NULL; IF a_pid.COUNT > 0 THEN FOR k IN a_pid.FIRST .. a_pid.LAST LOOP IF a_pid(k) = r.ITEM_ID THEN v_pi := k; EXIT; END IF; END LOOP; END IF; IF v_pi IS NULL THEN v_pi := v_pmax; a_pid(v_pi) := r.ITEM_ID; a_pname(v_pi) := r.PRODUCT_NAME; a_ptotal(v_pi):= 0; -- Pre-init all month cells for this product FOR m IN 0 .. v_mo_count - 1 LOOP a_bags(v_pi * 200 + m) := 0; END LOOP; v_pmax := v_pmax + 1; END IF; a_bags(v_pi * 200 + v_mi) := r.bags; a_ptotal(v_pi) := NVL(a_ptotal(v_pi), 0) + r.bags; a_motot(v_mi) := NVL(a_motot(v_mi), 0) + r.bags; v_gtot := v_gtot + r.bags; END LOOP; -- Print JavaScript for print functionality htp.p(''); htp.p(''); -- Report container htp.p('
'); htp.p('

Product Sales Bags Report — ' || TO_CHAR(v_start_date, 'MON-YYYY') || ' to ' || TO_CHAR(v_end_date, 'MON-YYYY') || '

'); htp.p('
Bag Type: ' || V_BAGS_TYPE || ' Print By: '|| :APP_USER ||'
'); htp.p(''); -- THEAD htp.p(''); htp.p(''); htp.p(''); -- Month columns in REVERSE order FOR i IN REVERSE 0 .. v_mo_count - 1 LOOP v_mo_date := ADD_MONTHS(v_start_date, i); htp.p(''); END LOOP; htp.p(''); -- TBODY: data rows (NO FOOTER HERE) htp.p(''); IF v_pmax > 0 THEN FOR pi IN 0 .. v_pmax - 1 LOOP htp.p(''); htp.p(''); htp.p(''); -- Monthly data for this product in REVERSE order FOR mi IN REVERSE 0 .. v_mo_count - 1 LOOP htp.p(''); END LOOP; htp.p(''); END LOOP; ELSE htp.p(''); END IF; htp.p(''); -- NOW ADD TOTAL ROW OUTSIDE OF FOOTER htp.p(''); htp.p(''); htp.p(''); -- Monthly totals in REVERSE order FOR mi IN REVERSE 0 .. v_mo_count - 1 LOOP htp.p(''); END LOOP; htp.p(''); htp.p('
PRODUCTTOTAL BAGS' || TO_CHAR(v_mo_date, 'MON-YYYY') || '
' || a_pname(pi) || '' || TO_CHAR(NVL(a_ptotal(pi), 0), '999,999,990') || '' || CASE WHEN NVL(a_bags(pi * 200 + mi), 0) = 0 THEN '–' ELSE TO_CHAR(a_bags(pi * 200 + mi), '999,999,990') END || '
No data found.
TOTAL' || TO_CHAR(v_gtot, '999,999,990') || '' || TO_CHAR(NVL(a_motot(mi), 0), '999,999,990') || '
'); htp.p('
'); END;
